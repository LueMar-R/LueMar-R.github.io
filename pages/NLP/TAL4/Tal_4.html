def pretraitement(text):
    #text = tokenizer.tokenize(text) #sépare les phrases et retourne une liste de phrases -> inutile ici
    text = text.lower() #lower case
    soup = BeautifulSoup(text, "html.parser") #suppression des balises html
    text = soup.get_text()
    text = re.sub('\[[^]]*\]', ' ', text)
    text = re.sub('[^a-zA-Z]', ' ', text)

    words = text.split(" ") # on remplace le tokenizer par un split, on obtient la liste des mots 
    words = list(filter(None, words))

    words = [word for word in words if word not in stop_words] #suppression des mots vides
    return words

Moyenne des vecteurs des mots existants pour un review donné

Chaque mot est représenté par un vecteur de dimension 300, on fait la moyenne de ces vecteurs pour une review donnée :

def moy_vect(sentence, model):
    total = np.zeros((model.vector_size,))
    count = 0
    for word in sentence : 
        try:
            add = model.wv[word]
            count += 1
            total += add
        except KeyError:
            continue
    mean = total/count
    return mean

def build_matrix(reviews_list, model):
    t = time.time()
    vectors_list = []
    for review in reviews_list : 
        vectors_list.append(moy_vect(review, model))
        
    vect_mat = []
    for v in vectors_list :
        vect_mat.append(v)
    vect_mat = np.asarray(vect_mat).reshape(len(vectors_list),model.vector_size)
    print('Time to build the matrix : {} secs'.format(round((time.time() - t), 2)))
    print("Matrix of shape", vect_mat.shape, "built succesfully")
    return vect_mat
